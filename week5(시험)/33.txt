mysql -u root -p
create database smddb;
use smddb;

mysql> create table st(
    -> sid int not null auto_increment,
    -> name varchar(30) not null,
    -> dept varchar(40),
    -> gpa float,
    -> year varchar(20),
    -> primary key(sid));

mysql> grant insert, select, update, delete, alter, create, drop
    -> on kimdb.* to 'jiho'@'localhost' identified by '1234';

    -> on kimdb.* to 'jiho'@'$' identified by '1234';
